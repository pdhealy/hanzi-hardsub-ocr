**tesseract.js initialization (per RESEARCH.md Pattern 2):**

The content script cannot use ES module imports for tesseract.js (it is a UMD bundle loaded via the content_scripts manifest entry or via script tag). Since we are using esbuild to bundle, we need to handle tesseract.js loading carefully.

Approach: Load the tesseract.js UMD bundle via a `<script>` tag injected into the page, which sets `window.Tesseract`. Then use `window.Tesseract.createWorker` from the content script.

However, content scripts run in an isolated world and cannot access `window.Tesseract` set by a page script. The correct approach for a bundled extension:

**Option A (recommended): Include tesseract.min.js as an additional content script in manifest.json.**
- This requires updating manifest.json to add `"libs/tesseract/tesseract.min.js"` BEFORE `"dist/content.bundle.js"` in the content_scripts js array.
- Then `Tesseract` will be available as a global in the content script's isolated world.
- In ocr.js, reference `Tesseract` (the global set by the UMD script) without importing it:
  ```javascript
  /* global Tesseract */
  ```

**If the UMD approach does not work** (verify during implementation by checking if `typeof Tesseract !== 'undefined'` after the script loads), fall back to:

**Option B: Use dynamic import or importScripts pattern.**
- Import tesseract.js ESM build if available, or use a dynamic `<script>` injection into the page's main world via `chrome.scripting.executeScript` with `world: 'MAIN'`.

**Start with Option A.** Update `extension/manifest.json` content_scripts to:
```json
"content_scripts": [
  {
    "matches": ["https://www.youtube.com/*"],
    "js": ["libs/tesseract/tesseract.min.js", "dist/content.bundle.js"],
    "run_at": "document_idle",
    "all_frames": false
  }
]
```

**Worker initialization:**
```javascript
async initialize() {
  if (this.worker) return;
  const workerPath = chrome.runtime.getURL('libs/tesseract/worker.min.js');
  const corePath = chrome.runtime.getURL('libs/tesseract-core/');
  const langPath = chrome.runtime.getURL('tessdata/');

  this.worker = await Tesseract.createWorker('chi_sim', 1, {
    workerPath,
    corePath,
    langPath,
    workerBlobURL: false,   // CRITICAL for MV3 CSP compliance
    gzip: false,            // traineddata is pre-decompressed
    logger: m => console.log('[YCR:Tesseract]', m.status, Math.round((m.progress || 0) * 100) + '%'),
  });
  this.initialized = true;
}
```

**Video frame capture (per RESEARCH.md Pattern 3):**
```javascript
async recognize(videoEl, intrinsicRect) {
  if (!this.worker) {
    await this.initialize();
  }

  // Create canvas and capture the selected region
  const canvas = document.createElement('canvas');
  canvas.width = intrinsicRect.width;
  canvas.height = intrinsicRect.height;
  const ctx = canvas.getContext('2d');

  // drawImage with source rectangle in video intrinsic pixel space
  ctx.drawImage(
    videoEl,
    intrinsicRect.x, intrinsicRect.y,
    intrinsicRect.width, intrinsicRect.height,
    0, 0,
    intrinsicRect.width, intrinsicRect.height
  );

  const dataURL = canvas.toDataURL('image/png');

  // Run OCR
  const result = await this.worker.recognize(dataURL, {}, { text: true });

  return {
    text: (result.data.text || '').trim(),
    confidence: result.data.confidence || 0,
  };
}
```

**Terminate:**
```javascript
async terminate() {
  if (this.worker) {
    await this.worker.terminate();
    this.worker = null;
    this.initialized = false;
  }
}
```

Rewrite `extension/content/content.js` to be the fully wired orchestrator connecting all three modules.

```javascript
import { SelectionOverlay } from './overlay.js';
import { SidePanel } from './sidepanel.js';
import { OCREngine } from './ocr.js';

let overlay = null;
let sidePanel = null;
let ocrEngine = null;

function ensureOverlay() {
  if (!overlay) overlay = new SelectionOverlay();
  return overlay;
}

function ensureSidePanel() {
  if (!sidePanel) sidePanel = new SidePanel();
  return sidePanel;
}

function ensureOCR() {
  if (!ocrEngine) ocrEngine = new OCREngine();
  return ocrEngine;
}

chrome.runtime.onMessage.addListener((message, sender, sendResponse) => {
  if (message.action === 'ACTIVATE_DRAW_MODE') {
    ensureOverlay().activateDrawMode();
    sendResponse({ ok: true });
    return;
  }

  if (message.action === 'GET_STATUS') {
    sendResponse({ boxDrawn: overlay ? overlay.hasSelection() : false });
    return;
  }

  if (message.action === 'SHOW_PANEL') {
    const panel = ensureSidePanel();
    panel.show();
    if (!overlay || !overlay.hasSelection()) {
      panel.showNoSelection();
    }
    sendResponse({ ok: true });
    return;
  }

  if (message.action === 'RECOGNIZE') {
    // Async handler — must return true to keep channel open
    (async () => {
      const panel = ensureSidePanel();
      panel.show();

      // Check if selection exists
      if (!overlay || !overlay.hasSelection()) {
        panel.showNoSelection();
        sendResponse({ ok: false, error: 'No selection' });
        return;
      }

      // Show loading state (per D-08)
      panel.showLoading();

      try {
        const engine = ensureOCR();
        const videoEl = document.querySelector('#movie_player video');
        if (!videoEl) {
          panel.showError('Video element not found');
          sendResponse({ ok: false, error: 'No video element' });
          return;
        }

        const intrinsicRect = overlay.getVideoIntrinsicRect();
        const result = await engine.recognize(videoEl, intrinsicRect);

        if (result.text && result.text.length > 0) {
          panel.showText(result.text);
          sendResponse({ ok: true, text: result.text });
        } else {
          // Per D-09: show "No text recognized", not blank
          panel.showEmpty();
          sendResponse({ ok: true, text: '' });
        }
      } catch (err) {
        console.error('[YCR] OCR error:', err);
        panel.showError(err.message || 'Unknown error');
        sendResponse({ ok: false, error: err.message });
      }
    })();
    return true; // CRITICAL: keeps message channel open for async response
  }
});

// YouTube SPA navigation cleanup (per RESEARCH.md Pitfall 5)
document.addEventListener('yt-navigate-finish', () => {
  if (overlay) { overlay.destroy(); overlay = null; }
  if (sidePanel) { sidePanel.destroy(); sidePanel = null; }
  if (ocrEngine) { ocrEngine.terminate(); ocrEngine = null; }
});

console.log('[YCR] YouTube Chinese Reader content script loaded');
```

Key wiring details:
1. ACTIVATE_DRAW_MODE: creates overlay, enters draw mode. Synchronous response.
2. GET_STATUS: reports whether a selection box exists. Synchronous response.
3. SHOW_PANEL: shows side panel, displays "No area selected" if no box drawn.
4. RECOGNIZE: the main flow — shows loading, runs OCR, displays result or empty/error state. Async — returns true to keep message channel open.
5. On `yt-navigate-finish`: destroy all modules and null references.
6. All error paths display user-facing messages in the side panel (never silent failures).

After writing, run `npm run build` to verify the full bundle compiles with all three module imports resolved.
